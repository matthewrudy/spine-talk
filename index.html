The Model

Its job

  • Access Data
  • Create Data
  • Persist Data

Solution:

Spine.Model

Define it!

var Message = 
  Spine.Model.setup(
  
    // the name
    "Message",
    
    // the fields
    ["body"
    ,"name"
    ,"channel_id"
    ]);

Add some Class Methods

Message.extend({
  poos: function(){
    return this.select(function(m){
      return m.body == "poo";
    });
  }
});

Message.poos()
// [Message, ...]

Default some attributes

Message.include({
  body: "blah blah blah",
  name: "Robot"
  });

Add some Instance Methods

Message.include({
  channel: function(){
    return Channel.find(
      this.channel_id
    );
  }
});

message.channel() // Channel:44

Validate it

Message.include({
  validate: function(){
    if(!this.channel_id){
      // return a message
      // to stop save()
      return "channel_id required";
    }
  }
});

Instantiate it!

var message = Message.init({
  name: "Peter",
  body: "oh gosh"
});

message.name // "Peter"
message.channel() // null
message.save() // Validation failed:

Persist in Memory!

message.channel_id = 12345;
message.save();

message.id; // "49557B45-7048-480C
            //  -BED8-23358E5239D4"

Message.all() == [message];

Persist locally!

// html 5 datastore

if(Modernizr.localstorage){
  Message.extend(
    Spine.Model.Local
  );
}

// loads from local storage
Message.fetch();

Persist via Ajax!

Message.extend(Spine.Model.Ajax);

// GET /messages
Message.fetch();

// POST /messages
var message = Message.create();

// PUT /messages/:id
message.save();

Override the URL!

Message.extend({
  url: "/chat/messages"
});

Catch errors

Message.bind("ajaxError",
  function(xhr, settings, error){
    showError(error);
  }
);

The View

Its job

  • Take some data
  • Display it

Solution:

jQuery.tmpl

"Static" HTML

Generated by your server

<div id="content">
  <div id="messages">
    <ul class="items">
    </ul>
  </div>
</div>

No data!

"Dynamic" HTML

Generated from template

<script id="messageTemplate"
  type="text/x-jquery-tmpl">
  <li class="item">
    <span class="name">
      ${name || "(unknown)"}
    </span>
    <span class="body">
      ${body}
    </span>
  </li>
</script>

Combine them!

var tmpl = $("#messageTemplate");
var list = $("#messages ul");

// add one message
list.append(tmpl.tmpl({
  name: "Steve",
  body: "Hello guys"
}));

// refresh them all
list.html(tmpl.tmpl(Message.all());

But actually its the Controller's job to coordinate View rendering...

The Controller

Its Job

  • User Interaction
  • User State
  • Keep the View upto date

Solution:

Spine.Controller

The List of Messages is a Controller

So is the List of Channels

And so is the Whole App!

I'll shut up and just talk about Messages

Actions

  • Show the List of Messages
  • Click on a Message to @reply
  • Input a Message and Click "send" to Create a Message

Define it!

var Messages = Spine.Controller.create({
  elements: {
    ".items": "items",
  },

  proxied: ["render"],
  
  render: function(){
    this.items.html(
      $("#messageTemplate").tmpl(
        Message.all()
      )
    );
  }
})

Instantiate it!

// use an existing DOM element
var messages = Messages.init(
  $("#messages")
);

// render it
messages.render();

Add the Create!

Messages = Spine.Controller.create({
  elements: {".new textarea": "input"},
  events:   {"click .new button": "create"},
  proxied:  ["create"],

  create: function(){
    Message.create({
      name:       "Peter",
      channel_id: 12345, 
      body: this.input.val();
    });
    
    this.input.val("");
    this.render();
  }
});

De-couple it!

Messages = Spine.Controller.create({
  init: function(){
    Message.bind("create", addOne);
    Message.bind("refresh", render);
  },
  create: function(){
    Message.create({
      name:       "Peter",
      channel_id: 12345, 
      body: this.input.val();
    });
    this.input.val("");
  },
  addOne: function(item){
    this.items.append(template(item));
  }
});
